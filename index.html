function agregarFila(usd = "", tasa = "") { const tbody = document.getElementById("tbody"); const fila = document.createElement("tr"); const cIndex = document.createElement("td"); cIndex.style.textAlign = "left"; cIndex.textContent = tbody.children.length + 1; const cUsd = document.createElement("td"); const inputUsd = document.createElement("input"); inputUsd.type = "number"; inputUsd.step = "0.01"; inputUsd.value = usd; inputUsd.oninput = recalcular; cUsd.appendChild(inputUsd); const cTasa = document.createElement("td"); const inputTasa = document.createElement("input"); inputTasa.type = "number"; inputTasa.step = "0.01"; inputTasa.value = tasa; inputTasa.oninput = recalcular; cTasa.appendChild(inputTasa); const cTotal = document.createElement("td"); cTotal.className = "total-cup"; cTotal.textContent = "0.00"; const cAccion = document.createElement("td"); const btnDel = document.createElement("button"); btnDel.textContent = "X"; btnDel.className = "btn-del"; btnDel.onclick = () => { fila.remove(); renumerar(); recalcular(); }; cAccion.appendChild(btnDel); fila.appendChild(cIndex); fila.appendChild(cUsd); fila.appendChild(cTasa); fila.appendChild(cTotal); fila.appendChild(cAccion); tbody.appendChild(fila); recalcular(); } function renumerar() { const filas = document.querySelectorAll("#tbody tr"); filas.forEach((tr, i) => tr.children[0].textContent = i + 1); } function recalcular() { const filas = document.querySelectorAll("#tbody tr"); let sumaUsd = 0; let sumaCup = 0; let sumaUsdPorTasa = 0; filas.forEach(tr => { const usd = parseFloat(tr.children[1].querySelector("input").value) || 0; const tasa = parseFloat(tr.children[2].querySelector("input").value) || 0; const totalCup = usd * tasa; tr.children[3].textContent = totalCup.toFixed(2); sumaUsd += usd; sumaCup += totalCup; sumaUsdPorTasa += usd * tasa; }); document.getElementById("totalUsd").textContent = sumaUsd.toFixed(2); document.getElementById("totalCup").textContent = sumaCup.toFixed(2); const promedio = sumaUsd > 0 ? (sumaUsdPorTasa / sumaUsd) : 0; document.getElementById("promedio").textContent = promedio.toFixed(2); }
